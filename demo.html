Implementation Code

Add this code to your website to integrate the Groq Website Assistant:

<!-- Groq Website Assistant -->
<script src="https://Groq-Website-Information-and-Sales-Assistant.vercel.app/src/chat-widget.js"></script>
<script>
  initGroqAssistant({
    apiEndpoint: 'https://Groq-Website-Information-and-Sales-Assistant.vercel.app/api/chat',
    companyName: 'Your Company',
    welcomeMessage: 'Hello! How can I help you today?',
    theme: 'dark', // or 'light'
    position: 'right', // or 'left'
    systemPrompt: `You are a helpful assistant for [Your Company].
    
    Company details:
    - Products: [Your products]
    - Pricing: [Your pricing]
    
    Your job is to be helpful and informative.`
  });
</script>
